type Theme = 'dark' | 'grey' | 'slate';
 
// Soft accents that stay readable for long sessions
const accents: Record<Theme, string> = {
  grey: '#ce9178',
  dark: '#2ec4c6',
  grey: '#e24285',
  slate: '#f4a82b',
};
 
function pickAccent(name: Theme): string {
  return accents[name];
}
